cleaningfunction<- function(corpus){
  remvURL <-function(x) gsub("http[^[:space:]]*", "", x)
  corpus <- tm_map(corpus, content_transformer(remvURL))
  
  remvusernames <- function(x) gsub("(^|[^@\\w])@(\\w{1,15})\\b", "", x)
  corpus <- tm_map(corpus, content_transformer(remvusernames)) 
   
  corpus <- tm_map(corpus, content_transformer(tolower))
  
  corpus <- tm_map(corpus, removePunctuation) 
  
  corpus <- tm_map(corpus, removeNumbers)
  
  customstopwords <- c(stopwords("spanish"), "rt", "retweet", "amp", "follow", "\n")
  
  corpus <- tm_map(corpus, removeWords, customstopwords)
 
  corpus <- tm_map(corpus, stripWhitespace)
  return(corpus)
}

#Training

Now I can start training and testing two different algorithms to obtain two supervised classification models, compare them and choose the one with the best results.

#Setting the train performance control method and the metric
ctrl <- trainControl(method="cv", number = 10)
metric = "Accuracy"
#Training with Naive-Bayes
#install.packages("naivebayes")
training$type <- as.factor(training$type) #code used to resolve the error at different levels 
nbModel <- train(type ~ ., data=training,method="naive_bayes", tuneLength=10, trControl=ctrl , metric=metric)
